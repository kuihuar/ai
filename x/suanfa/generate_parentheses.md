### 22. 括号生成

数字 n 代表生成括号的对数，请你设计一个函数，用于能够生成所有可能的并且 有效的 括号组合。
示例 1：
输入：n = 3
输出：["((()))","(()())","(())()","()(())","()()()"]
示例 2：
输入：n = 1
输出：["()"]
示例 3：
输入：n = 2
输出：["(())","()()"]
提示：
```go
func GenerateParenthesis(n int) []string {
	var result []string
	var dfs func(left, right int, path string)
	dfs = func(left, right int, path string) {
		if left == n && right == n {
			fmt.Println("dfs(%d, %d, %s)返回，添加到结果集\n", left, right, path)
			result = append(result, path)
			return
		}

		if left < n {
            fmt.Println("dfs(%d+1, %d, %s+\"(\")\n", left, right, path) 
			dfs(left+1, right, path+"(")
		}
		if right < left {
            fmt.Println("dfs(%d, %d+1, %s+\")\")\n", left, right, path) 

			dfs(left, right+1, path+")")
		}
	}
	dfs(0, 0, "")
	return result
}
